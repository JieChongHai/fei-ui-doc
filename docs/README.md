# 快速开始

> 在 main.js 中引入组件库

```javascript
// 全部引入
import 'feifei-ui/dist/css/index.css';
import FeiUI from 'feifei-ui';
Vue.use(FeiUI);

// 按需引用
import 'feifei-ui/dist/css/card.css';
import { card } from 'feifei-ui';
Vue.use(card);
```
